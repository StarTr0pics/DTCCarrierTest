    Data interchange format between Server and Runtime (how data is stored inside runtime core)

    SD Runtime fields data:

    fieldName: {
    value: 'val',
    formattedValue: 'formattedval'
}

anotherFieldName: {
    value: '123'
    formattedValue: '$123'
}

    Corresponding Server fields data:

    ...
fieldName = 'val',
fieldName_fmt = 'formattedval',
anotherFieldName = '123',
anotherFieldName_fmt = '$123',
...

    SD Runtime SourceDropdown control data:

    // Single select dropdown:
sourceDropdownName: {
    value: 'selectedOptionValue',
    formattedValue: 'selectedOptionName'
    data: {
        options: [{
            opt1: "opt1value",
            anotheropt: "anotherValue"
        }
    ]}
}


// Multiselect dropdown (with 3 selected values):

multiSourceDropdownName: {
    value: 'val1,val2,val3',
    formattedValue: 'name1,name2,name3'
    data: {
        options: [
            {
                opt1: "opt1value",
                anotheropt: "anotherValue"
            },
            {
                somethig: "someOtherValue"
            },
            {
                // no additional options
            }
        ]
    }
}

    Corresponding Server SourceDropdown control data:

    cases:

    // Single select dropdown:
...
sourceDropdownName = 'selectedOptionValue',
sourceDropdownName_fmt = 'selectedOptionName',
sourceDropdownName||options = '[{"name":"selectedOptionName", "value":"selectedOptionValue", "opt1":"opt1value", "anotheropt":"anotherValue"}]'
...

// Multiselect dropdown (with 3 selected values):
...
multiSourceDropdownName = 'val1,val2,val3',
multiSourceDropdownName_fmt = 'name1|name2|name3',
multiSourceDropdownName||options = '[{"name":"val1", "value":"name1", "opt1":"opt1value", "anotheropt":"anotherValue"}, {"name":"val2", "value":"name2", "something":"someOtherValue"}, {"name":"val3", "value":"name3"}]'
...

NOTES:
1. "fieldName__options" key contains double-underscore"
2. "fieldName__options" should contain valid JSON in a string
3. "fieldName__options" should contain all data for a selected option, including "name" and "value" despite they are already available in fieldName and fieldName_fmt.

    SD Runtime Grid control data:

    gridName: {
    data: [{                       // first row
        col1: {
            value: 'john',
            formattedValue: 'johnny'
        },
        col2: {
            value: '123',
            formattedValue: '$123'
        }
    },{                            // second row
        col1: {
            value: 'don',
            formattedValue: 'donny'
        },
        col2: {
            value: '9',
            formattedValue: '$9'
        }
    }]
}

    Corresponding Server Grid control data:

    // NOTE: grid name is not used!
...,
col1__1 = 'john',
col1__1_fmt = 'johnny',
col2__1 = '123',
col2__1_fmt = '$123',
col1__2 = 'don',
col1__2_fmt = 'donny',
col2__2 = '9',
col2__2_fmt = '9',
...

    SD Runtime SourceGrid control data:

    sourceGridName: {
    data: [{                        // first row
        "row-id": {                 // "row-id" is predefined name
            value: 'id1',           // "row-id" values are unique across source grid rows
            formattedValue: ''
        },
        col1: {
            value: 'john',
        },
        col2: {
            value: '123',
            formattedValue: '$123'
        }
    },{                             // second row
        "row-id": {
            value: 'id2',
                formattedValue: ''
        },
        col1: {
            value: 'don',
        },
        col2: {
            value: '9',
            formattedValue: '$9'
        }
    }]
}

    Corresponding Server SourceGrid control data:

    // NOTE: grid name is not used!
...,
sourceGridName = '[{"row-id":{"value":"id1","formattedValue":""},"col1":{"value":"john"},"col2":{"value":"123","formattedValue":"$123"}},{"row-id":{"value":"id1","formattedValue":""},"col1":{"value":"don"},"col2":{"value":"9","formattedValue":"$9"}}]'
...

    SD Runtime Mask control data:

    maskcontrolName: {
    value: 'value',
    formattedValue: 'formatted-value',
    data: {
        maskedValue: 'masked-value'
    }
}

    Corresponding Server Mask control data:

    cases:

    // properties for maskcontrolName 'showMaskedCharactersInPDF' and 'encrypt' = false

maskcontrolName = 'value',
maskcontrolName_fmt = 'formatted-value',
maskcontrolName_enc = false
// NOTE: maskcontrolName_msk doesn't added

// 'showMaskedCharactersInPDF' and 'encrypt' = true

maskcontrolName = 'value',
maskcontrolName_fmt = 'formatted-value',
maskcontrolName_msk = 'masked-value',
maskcontrolName_enc = true

    SD Runtime SoftLocked data:

    name: {
    value: 'value',

lockedName: {
    value: 'value',
    formattedValue: 'formatted-value',
    locked: true
}

// in case when control empty and belongs to softLocked screen:
emptyLockedName: {
    locked: true
}

    data: [{
        col1: {
            value: 'john',
            formattedValue: 'johnny'
        },
        col2: {
            value: '123',
            formattedValue: '$123'
        }
    }],
    locked: true
}
