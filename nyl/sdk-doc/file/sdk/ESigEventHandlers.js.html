    import _ from 'lodash';
import logger from 'runtime/../../logger';
import moment from 'shared/moment';
import createESIGViewPDFModalWindow from 'runtime/sdk/components/ESIGViewPDFModalWindow.react';
import * as ft from 'shared/constants/FieldTypes';
import { FINISH_ESIG_UI_INTERNAL } from 'runtime/constants/MessageTypes';
import postMessageHelper from 'runtime/utils/postMessageHelper';

const _addCapturedEvent = (context, data) => {
    if (!_.isArray(context.capturedEvents)) context.capturedEvents = [];

    if (!_.isPlainObject(data)) data = {};

    const { fieldName = '', fieldType = '', fieldValue = '', eventType = '' } = data;

    const event = {
        FieldName: fieldName,
        FieldType: fieldType,
        EventDate: moment.utc().format()
    };

    if (fieldType === ft.BUTTON) {
        event.FieldValue = '';
        event.EventType = 'click';
    } else if (fieldType === ft.RADIOBUTTON) {
        event.FieldValue = fieldValue;
        event.EventType = 'click';
    } else if (fieldType === ft.CHECKBOX) {
        if (fieldValue === 'true') event.EventType = 'checked';
        if (fieldValue === 'false') event.EventType = 'unchecked';
        event.FieldValue = '';
    } else {
        event.FieldValue = fieldValue;
        event.EventType = eventType;
    }

    context.capturedEvents.push(event);
};

const _getCapturedEvents = context =>
    _.isArray(context.capturedEvents) ? context.capturedEvents : [];

const _clearCapturedEvents = context => {
    context.capturedEvents = [];
};

/**
 * eSig UI SDK class. Used as separate SDK class for eSig UI applications.
 */
export class ESigEventHandlers {
    /**
     * This method is invoked whenever runtime executes the "capture event" action in eSigUI
     * @since v0.89.*
     * @param {Object} data - event data
     * @param {String} data.fieldName - name of a field that generated the event
     * @param {String} data.fieldType - type of a field that generated the event
     * @param {String} data.fieldValue - value of a field that generated the event
     * @param {String} data.eventType - type of the event on a field that generated the capture event (e.g. 'click', 'valuechanged', etc.)
     */
    async ESIGUI_captureEvent(data) {
        _addCapturedEvent(this, data);
    }

    async ESIGUI_flushAuditTrail() {
        const serviceRequestProvider = this.coreApi.serviceRequestProvider;
        const clientDataKey = this.coreApi.runtimeStateItems.CLIENT_DATA;
        const capturedEvents = _getCapturedEvents(this);

        const authToken = await this.requestAuthToken();

        const { success, error, message } = await serviceRequestProvider.ESIGUI_flushAuditTrail(
            capturedEvents,
            authToken
        );

        this.coreApi.updateRuntimeState({
            [clientDataKey]: {
                ESIGUI_flushAuditTrail_isSuccessful: { value: success, formattedValue: success },
                ESIGUI_flushAuditTrail_errorCode: { value: error, formattedValue: error },
                ESIGUI_flushAuditTrail_errorMessage: { value: message, formattedValue: message }
            }
        });

        _clearCapturedEvents(this);
    }

    /**
     * This method is invoked whenever runtime executes the "accept disclosure" action in eSigUI.
     * @since v0.89.*
     */
    async ESIGUI_acceptDisclosure() {
        const serviceRequestProvider = this.coreApi.serviceRequestProvider;
        const clientDataKey = this.coreApi.runtimeStateItems.CLIENT_DATA;

        const authToken = await this.requestAuthToken();

        const { success, error, message } = await serviceRequestProvider.ESIGUI_acceptDisclosure(
            authToken
        );

        this.coreApi.updateRuntimeState({
            [clientDataKey]: {
                ESIGUI_disclosureAcceptance_isSuccessful: {
                    value: success,
                    formattedValue: success
                },
                ESIGUI_disclosureAcceptance_errorCode: { value: error, formattedValue: error },
                ESIGUI_disclosureAcceptance_errorMessage: {
                    value: message,
                    formattedValue: message
                }
            }
        });
    }

    /**
     * This method is invoked whenever runtime executes the "view PDF" action in eSigUI (not to be confused with generic "view PDF", which is not eSig-specific)
     * @since v0.89.*
     */
    async ESIGUI_viewPDF() {
        const authToken = await this.requestAuthToken();

        const { data } = await this.coreApi.serviceRequestProvider.ESIGUI_getPDFsURL(authToken);
        const url = _.get(data, ['Signers', 0, 'DocumentViewUrl'], null);

        if (!url) throw new Error('ESIGUI: Url for viewing document (pdf) was not provided');

        this.coreApi.showModal(
            createESIGViewPDFModalWindow({ pdfUrl: url, onClose: this.coreApi.hideModal })
        );
    }

    /**
     * This method is invoked whenever runtime executes the "get signing status" action in eSigUI (after the nested application receives a post message about signature completion)
     * @since v0.89.*
     * @returns {Promise<Object>} - promise of signing status
     */
    async ESIGUI_getSigningStatus() {
        const ANY_ERROR_STATUS = -1;
        const serviceRequestProvider = this.coreApi.serviceRequestProvider;
        const clientDataKey = this.coreApi.runtimeStateItems.CLIENT_DATA;

        const authToken = await this.requestAuthToken();

        const response = await serviceRequestProvider.ESIGUI_getSigningStatus(authToken);
        const { data, success, error, message } = response;
        const status = _.get(data, ['DynamicData', 'StatusId'], ANY_ERROR_STATUS);

        this.coreApi.updateRuntimeState({
            [clientDataKey]: {
                ESIGUI_vendorSigning_status: { value: status, formattedValue: status },
                ESIGUI_vendorSigning_isSuccessful: { value: success, formattedValue: success },
                ESIGUI_vendorSigning_errorCode: { value: error, formattedValue: error },
                ESIGUI_vendorSigning_errorMessage: { value: message, formattedValue: message }
            }
        });

        return status;
    }

    /**
     * This method is invoked whenever runtime executes the "get vendor URL" action in eSigUI
     * @since v0.89.*
     * @returns {Promise<String>} - promise of the vendor URL
     */
    async ESIGUI_getVendorURL() {
        const serviceRequestProvider = this.coreApi.serviceRequestProvider;
        const clientDataKey = this.coreApi.runtimeStateItems.CLIENT_DATA;

        const authToken = await this.requestAuthToken();

        const response = await serviceRequestProvider.ESIGUI_getVendorURL(authToken);
        const { data, success, error, message } = response;

        this.coreApi.updateRuntimeState({
            [clientDataKey]: {
                ESIGUI_requestURL_isSuccessful: { value: success, formattedValue: success },
                ESIGUI_requestURL_errorCode: { value: error, formattedValue: error },
                ESIGUI_requestURL_errorMessage: { value: message, formattedValue: message }
            }
        });

        return _.get(data, ['Signers', 0, 'SigningUrl'], null);
    }

    /**
     * This method is invoked when nested application Runtime attempts to initialize with external data that has been transmitted from host application.
     * @since v0.92.*
     * @param {Object} data - ClientData from the host application, see format description in `SD Runtime fields data`;
     * @returns {undefined}
     */
    initialLoadData(data) {
        if (!data) return;

        this.coreApi.restoreRuntimeState({
            [this.coreApi.runtimeStateItems.CLIENT_DATA]: data
        });
    }

    /**
     * This method is invoked whenever runtime executes the "finish signing" action in eSigUI
     * @since v0.92.*
     */
    async ESIGUI_finishSigning() {
        postMessageHelper.sendMessageToParent({ type: FINISH_ESIG_UI_INTERNAL });
    }

    /**
     * This helper method knows how to get authentication token for requests to esig service.
     * @since v0.97.*
     * @returns {string}
     */
    async requestAuthToken() {
        const serviceRequestProvider = this.coreApi.serviceRequestProvider;
        let authToken;

        const maxNumberOfAttempts = 10; // TODO: take from config
        let counter = 0;
        while (!authToken && ++counter <= maxNumberOfAttempts) {
            try {
                authToken = await serviceRequestProvider.getEsigUIServiceAuthToken();
            } catch (e) {
                logger.runtime.coreApi.error(
                    e,
                    `failed to request ESIG authToken, attempt - ${counter}`
                );
            }
        }

        return authToken;
    }
}
